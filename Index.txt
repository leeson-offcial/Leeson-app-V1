// ── D-Day 계산 ────────────────────────────────
function calcDDay(dateStr) {
  const today=new Date(); today.setHours(0,0,0,0);
  const target=new Date(dateStr); target.setHours(0,0,0,0);
  return Math.ceil((target-today)/(1000*60*60*24));
}

// ── Gemini 직접 호출 ──────────────────────────
async function callGemini(userText) {
  const body = {
    system_instruction: { parts: [{ text: getSystemPrompt() }] },
    contents: [{ role: "user", parts: [{ text: userText }] }],
    generationConfig: { temperature: 0.85, maxOutputTokens: 8192 }
  };
  const res = await fetch(GEMINI_URL, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body)
  });
  const json = await res.json();
  if (json.error) throw new Error(json.error.message);
  return json.candidates[0].content.parts[0].text;
}

  }).catch(()=>{});
}

// ── 식단 시작 ─────────────────────────────────
async function startDiet() {
  if (!validate()) return;
  saveState();

  const p = {
    name: val("f_name").trim(), age: val("f_age"), gender: state.gender,
    job: val("f_job").trim(), targetDate: val("f_targetDate"),
    currentWeight: val("f_currentWeight"), targetWeight: val("f_targetWeight"),
    purpose: val("f_purpose").trim(), exercise: val("f_exercise").trim(),
    water: val("f_water").trim(), notes: val("f_notes").trim(),
  };

  showLoading("맞춤 식단 생성 중... 🥗");
  try {
    const dday = calcDDay(p.targetDate);
    const tier = getTier(dday);
    const userId = makeUserId(p.name);
    const diet = await callGemini(buildDietPrompt(p, dday, tier));

    state.userId=userId; state.dday=dday; state.tier=tier;
    state.name=p.name; state.currentWeight=p.currentWeight;
    state.targetWeight=p.targetWeight; state.notes=p.notes;
    state.dietRaw=diet;
    saveState();

    // 구글 시트 기록 (백그라운드)
    logToSheet("generateDiet", { ...p, userId, dday, tier, diet });

    renderDiet({ dday, tier, diet });
    goStep(3);
  } catch(e) {
    showToast("오류: " + e.message);
  } finally { hideLoading(); }
}

async function sendChat() {
  const input = document.getElementById("chat-input");
  const msg = input.value.trim();
  if (!msg) return;
  if (!state.userId) { showToast("먼저 식단을 시작해주세요!"); goStep(1); return; }

  input.value=""; input.style.height="auto";
  state.chatTurn++;
  appendMsg("user", msg);
  appendMsg("ai","···");
  const typingEl = document.getElementById("chat-messages").lastElementChild;

  try {
    const payload = {
      userId:state.userId, name:state.name,
      currentWeight:state.currentWeight, targetWeight:state.targetWeight,
      dday:state.dday, tier:state.tier, notes:state.notes,
      message:msg, turn:state.chatTurn,
      history:state.chatHistory.slice(-6),
    };
    const reply = await callGemini(buildChatPrompt(payload));
    typingEl.remove();
    appendMsg("ai", reply);
    state.chatHistory.push({user:msg, ai:reply});
    saveState();
    renderHistory();
    logToSheet("chat", { ...payload, reply });
  } catch(e) {
    typingEl.remove();
    appendMsg("ai","오류: "+e.message);
  }
}

function markdownToHtml(md) {
  md = md.replace(/((^\|.+\|[ \t]*\n)+)/gm, block => {
    const lines=block.trim().split("\n");
    if(lines.length<2) return block;
    const sepIdx=lines.findIndex(l=>/^\|[\s\-:|]+\|/.test(l));
    if(sepIdx<1) return block;
    const parse=line=>line.split("|").map(c=>c.trim()).filter((_,i,a)=>i>0&&i<a.length-1);
    const ths=parse(lines[sepIdx-1]).map(c=>`<th>${c}</th>`).join("");
    const trs=lines.slice(sepIdx+1).map(line=>{
      const tds=parse(line).map(c=>`<td>${c||"&nbsp;"}</td>`).join("");
      return tds?`<tr>${tds}</tr>`:"";
    }).filter(Boolean).join("");
    return `<div class="table-wrap"><table><thead><tr>${ths}</tr></thead><tbody>${trs}</tbody></table></div>\n`;
  });
  md=md
    .replace(/^#{1,3} (.+)$/gm,'<strong class="section-title">$1</strong>')
    .replace(/\*\*(.+?)\*\*/g,"<strong>$1</strong>")
    .replace(/\n{2,}/g,"</p><p>")
    .replace(/\n/g,"<br>");
  return `<p>${md}</p>`;
}
